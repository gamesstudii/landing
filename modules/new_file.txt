    <script>
        document.querySelectorAll('.nav-btn').forEach(btn => {
            btn.addEventListener('click', function(e) {
                e.preventDefault();
                
                document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                
                const pageId = this.getAttribute('href').substring(1);
                document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
                document.getElementById(pageId).classList.add('active');
            });
        });

        function toggleSmooth(element) {
            const subtopics = element.nextElementSibling;
            if (subtopics.style.maxHeight) {
                subtopics.style.maxHeight = null;
            } else {
                subtopics.style.maxHeight = subtopics.scrollHeight + "px";
            }
        }

        function toggleSubSmooth(element) {
            const subSubtopics = element.nextElementSibling;
            if (subSubtopics.style.maxHeight) {
                subSubtopics.style.maxHeight = null;
            } else {
                subSubtopics.style.maxHeight = subSubtopics.scrollHeight + "px";
            }
        }
    </script>
    <script>
    function checkAge() {
        document.getElementById('ageModal').style.display = 'block';
        const messageDiv = document.getElementById('message');
        if (messageDiv) {
            messageDiv.style.display = 'none';
        }
    }

    function verifyAge() {
        const ageInput = document.getElementById('ageInput');
        const age = parseInt(ageInput.value);
        const messageDiv = document.getElementById('message');

        if (isNaN(age) || age <= 0 || age > 120) {
            messageDiv.textContent = 'Пожалуйста, введите корректный возраст!';
            messageDiv.style.background = '#5c2a2a';
            messageDiv.style.color = '#ff8a8a';
            messageDiv.style.display = 'block';
            return;
        }

        if (age >= 18) {
            window.location.href = 'https://scratch.mit.edu/projects/1190285060';
        } else {
            messageDiv.textContent = 'Возраст недостаточен. Вам должно быть 18 лет или больше.';
            messageDiv.style.background = '#5c2a2a';
            messageDiv.style.color = '#ff8a8a';
            messageDiv.style.display = 'block';
            
            setTimeout(() => {
                document.getElementById('ageModal').style.display = 'none';
                ageInput.value = '';
            }, 3000);
        }
    }

    document.getElementById('ageInput')?.addEventListener('keypress', function(e) {
        if (e.key === 'Enter') {
            verifyAge();
        }
    });
    </script>
